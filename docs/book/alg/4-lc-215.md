## 快排

```cpp
class Solution {
    public int findKthLargest(int[] nums, int k) {
        int len = nums.length;
        quick_sort(nums , 0, len - 1);
        return nums[len - k];
    }

    void quick_sort(int[] q, int l, int r) {
        if (l >= r) return; // 切记不是 > ，而是 >= 
        int x = q[l + r >> 1];
        int i = l - 1, j = r + 1;  // 因为 do while
        while (i < j) // 切记没有等于
        {
            do i ++ ; while (q[i] < x);
            do j -- ; while (q[j] > x);
            if (i < j) {
                int t = q[i];
                q[i] = q[j];
                q[j] = t;
            }
        }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
    }
}
```


## 快排模板

```cpp
void quick_sort(int q[], int l, int r) {
    if (l >= r) return; // 切记不是 > ，而是 >= 
    int x = q[l + r >> 1];
    int i = l - 1, j = r + 1;  // 因为 do while
    while (i < j) // 切记没有等于
    {
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}
```
